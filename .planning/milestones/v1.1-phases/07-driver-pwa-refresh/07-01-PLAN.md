**Add new JS functions (place after the STORAGE_KEYS section):**

```javascript
function updateProgressBar() {
    if (!currentRoute) return;
    const bar = document.getElementById('progress-bar');
    const nextIdx = currentRoute.stops.findIndex(s => s.status === 'pending');
    bar.innerHTML = currentRoute.stops.map((stop, i) => {
        let cls = 'progress-pending';
        if (stop.status === 'delivered') cls = 'progress-delivered';
        else if (stop.status === 'failed') cls = 'progress-failed';
        else if (i === nextIdx) cls = 'progress-current';
        return `<div class="progress-segment ${cls}"></div>`;
    }).join('');
}

function updateHeaderStats() {
    if (!currentRoute) return;
    const done = currentRoute.stops.filter(s => s.status === 'delivered').length;
    const total = currentRoute.stops.length;
    document.getElementById('header-stats').textContent = `${done} of ${total} delivered`;
}

function updateRefreshRow() {
    const ts = loadFromStorage(STORAGE_KEYS.LAST_UPDATED);
    const el = document.getElementById('last-updated');
    if (ts && el) {
        const d = new Date(ts);
        el.textContent = `Last updated: ${d.toLocaleTimeString([], { hour: '2-digit', minute: '2-digit' })}`;
    }
}
```

function checkAllDone() {

    if (allDone && !existing) {
