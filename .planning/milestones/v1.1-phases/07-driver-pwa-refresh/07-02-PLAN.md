```javascript
function renderStopList() {
    const container = document.getElementById('stop-list');
    if (!currentRoute || !currentRoute.stops) {
        container.innerHTML = '';
        return;
    }

    const nextStop = currentRoute.stops.find(s => s.status === 'pending');
    const otherStops = currentRoute.stops.filter(s => s !== nextStop);

    container.innerHTML = [
        nextStop ? renderHeroCard(nextStop) : '',
        ...otherStops.map(s => renderCompactCard(s)),
    ].join('');

    updateProgressBar();
    updateHeaderStats();
    checkAllDone();
}
```

**Add `renderHeroCard()` function:**
```javascript
function renderHeroCard(stop) {
    const stopIdx = currentRoute.stops.indexOf(stop);
    const total = currentRoute.stops.length;
    const distInfo = stop.distance_from_prev_km > 0
        ? `${stop.distance_from_prev_km} km away`
        : '';
    const refInfo = stop.customer_ref
        ? `<div class="hero-meta">Ref: ${escapeHtml(stop.customer_ref)}</div>`
        : '';
    return `
    <div class="hero-card" id="stop-${escapeHtml(stop.order_id)}">
        <div class="hero-label">NEXT DELIVERY &middot; Stop ${stop.sequence} of ${total}</div>
        <div class="hero-address">${escapeHtml(stop.address) || 'Address pending'}</div>
        <div class="hero-meta">
            ${stop.quantity} cyl &middot; ${stop.weight_kg} kg${distInfo ? ` &middot; ${distInfo}` : ''}
        </div>
        ${refInfo}
        ${stop.notes ? `<div class="hero-notes">${escapeHtml(stop.notes)}</div>` : ''}
        <div class="hero-actions">
            <button class="btn btn-navigate" onclick="navigateTo(${stop.latitude}, ${stop.longitude})">
                &#x2316; Navigate
            </button>
            <div class="btn-row">
                <button class="btn btn-deliver" onclick="markDelivered('${escapeHtml(currentRoute.vehicle_id)}', '${escapeHtml(stop.order_id)}')">
                    &#x2713; Done
                </button>
                <button class="btn btn-fail" onclick="markFailed('${escapeHtml(currentRoute.vehicle_id)}', '${escapeHtml(stop.order_id)}')">
                    &#x2715; Fail
                </button>
            </div>
        </div>
    </div>`;
}
```

```javascript
function renderCompactCard(stop) {
    const statusClass = escapeHtml(stop.status);
    const isPending = stop.status === 'pending';
    return `
    <div class="compact-card ${statusClass}" id="stop-${escapeHtml(stop.order_id)}">
        <div class="stop-number">
            <div class="stop-number-inner">${stop.sequence}</div>
        </div>
        <div class="stop-info">
            <div class="stop-address">${escapeHtml(stop.address) || 'Address pending'}</div>
            <div class="stop-meta">
                ${stop.quantity} cyl &middot; ${stop.weight_kg} kg
                ${stop.distance_from_prev_km > 0 ? ` &middot; ${stop.distance_from_prev_km} km` : ''}
            </div>
            ${!isPending ? `<div class="status-chip ${statusClass}">
                ${stop.status === 'delivered' ? '&#x2713; Delivered' : '&#x2715; Failed'}
            </div>` : ''}
        </div>
    </div>`;
}
```

```javascript
function showToast(message, type) {
    // Remove any existing toast
    document.querySelector('.toast')?.remove();
    const toast = document.createElement('div');
    toast.className = `toast toast-${type}`;
    toast.textContent = message;
    document.body.appendChild(toast);
    setTimeout(() => toast.remove(), 1500);
}
```

```javascript
async function updateStatus(vehicleId, orderId, status) {
    const stop = currentRoute.stops.find(s => s.order_id === orderId);
    if (stop) stop.status = status;
    saveToStorage(STORAGE_KEYS.ROUTE_DATA, currentRoute);

    // Show toast feedback immediately
    showToast(
        status === 'delivered' ? 'Delivered!' : 'Marked failed',
        status
    );

    // Delay re-render for toast visibility, then auto-advance
    setTimeout(() => {
        renderStopList();
    }, 1500);

    // Update progress bar and header stats immediately (no delay)
    updateProgressBar();
    updateHeaderStats();

    try {
        const resp = await authPost(
            `${API_BASE}/api/routes/${vehicleId}/stops/${orderId}/status`,
            { status },
        );
        if (!resp.ok) throw new Error('Server rejected update');
    } catch {
        queueOfflineUpdate(vehicleId, orderId, status);
        console.log(`[Offline] Queued status update: ${orderId} → ${status}`);
    }
}
```
