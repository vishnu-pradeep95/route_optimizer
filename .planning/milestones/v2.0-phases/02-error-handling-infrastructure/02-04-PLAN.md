    **Test structure:**

    ```typescript
    import { test, expect } from '@playwright/test';
    import { validateApiKey } from './helpers/setup';

    test.describe.configure({ mode: 'serial' });

    test.describe('Dashboard Error UI', () => {
      test.beforeAll(async ({ request }) => {
        await validateApiKey(request);
      });

      // ─── ErrorBanner Tests ───────────────────────────────────

      test('ErrorBanner renders with correct severity on upload error', async ({ page }) => {
        // Navigate to dashboard upload page
        await page.goto('/dashboard/');
        // Wait for React SPA to load
        await page.waitForLoadState('networkidle');

        // Trigger an error by uploading an invalid file type
        // Create a fake .txt file and upload it via the file input
        const fileInput = page.locator('input[type="file"]');
        // If file input is hidden, trigger it via the upload button
        await page.evaluate(() => {
          const input = document.querySelector('input[type="file"]') as HTMLInputElement;
          if (input) {
            const file = new File(['not a csv'], 'test.txt', { type: 'text/plain' });
            const dt = new DataTransfer();
            dt.items.add(file);
            input.files = dt.files;
            input.dispatchEvent(new Event('change', { bubbles: true }));
          }
        });

        // Wait for error banner to appear (API returns UPLOAD_INVALID_FORMAT)
        const banner = page.locator('[data-testid="error-banner"]');
        await expect(banner).toBeVisible({ timeout: 10_000 });

        // Verify severity color: should be "error" (red) for upload failures
        await expect(banner).toHaveAttribute('data-severity', 'error');

        // Verify the banner contains tw:alert-error class
        await expect(banner).toHaveClass(/tw:alert-error|tw\\:alert-error/);

        // Verify user_message text is visible (not empty)
        const messageText = banner.locator('p').first();
        await expect(messageText).not.toBeEmpty();
      });

      test('ErrorBanner "Show details" toggle reveals error_code, request_id, timestamp', async ({ page }) => {
        await page.goto('/dashboard/');
        await page.waitForLoadState('networkidle');

        // Trigger an error (same approach as above)
        await page.evaluate(() => {
          const input = document.querySelector('input[type="file"]') as HTMLInputElement;
          if (input) {
            const file = new File(['not a csv'], 'test.txt', { type: 'text/plain' });
            const dt = new DataTransfer();
            dt.items.add(file);
            input.files = dt.files;
            input.dispatchEvent(new Event('change', { bubbles: true }));
          }
        });

        const banner = page.locator('[data-testid="error-banner"]');
        await expect(banner).toBeVisible({ timeout: 10_000 });

        // Click "Show details" toggle
        const detailToggle = page.locator('[data-testid="error-detail-toggle"]');
        await expect(detailToggle).toBeVisible();
        await detailToggle.click();

        // Verify detail panel is visible
        const detailPanel = page.locator('[data-testid="error-detail"]');
        await expect(detailPanel).toBeVisible();

        // Verify error_code is displayed (contains "UPLOAD_" prefix)
        await expect(detailPanel).toContainText(/UPLOAD_|ERROR/);

        // Verify request_id is displayed (8 hex chars pattern)
        // Just check for "Request ID:" label presence
        await expect(detailPanel).toContainText(/Request ID/i);

        // Verify timestamp is displayed
        await expect(detailPanel).toContainText(/Time/i);
      });

      test('ErrorBanner Retry button is visible and clickable', async ({ page }) => {
        await page.goto('/dashboard/');
        await page.waitForLoadState('networkidle');

        // Trigger an error
        await page.evaluate(() => {
          const input = document.querySelector('input[type="file"]') as HTMLInputElement;
          if (input) {
            const file = new File(['not a csv'], 'test.txt', { type: 'text/plain' });
            const dt = new DataTransfer();
            dt.items.add(file);
            input.files = dt.files;
            input.dispatchEvent(new Event('change', { bubbles: true }));
          }
        });

        const banner = page.locator('[data-testid="error-banner"]');
        await expect(banner).toBeVisible({ timeout: 10_000 });

        // Verify Retry button exists and is clickable
        const retryBtn = banner.getByRole('button', { name: /retry/i });
        await expect(retryBtn).toBeVisible();
        await expect(retryBtn).toBeEnabled();

        // Click retry -- should not crash (may re-trigger the error, that's fine)
        await retryBtn.click();
        // Page should still be functional after retry
        await expect(page.locator('body')).toBeVisible();
      });

      // ─── ErrorTable Tests ───────────────────────────────────

      test('ErrorTable renders rows with correct columns after CSV with bad rows', async ({ page, request }) => {
        // Upload a CSV with intentionally bad rows via API first,
        // then navigate to dashboard to see the error table.
        // Alternative: create a test CSV inline and upload via the UI.

        await page.goto('/dashboard/');
        await page.waitForLoadState('networkidle');

        // Create a CSV with a valid header but some bad data rows
        // The CSV must have the expected columns but with invalid content
        // that will fail validation or geocoding
        const badCSV = [
          'Order ID,Customer Name,Address,Phone,Status,Quantity',
          'ORD-001,Test Customer,Valid Address Trivandrum Kerala,9876543210,Allocated-Printed,1',
          ',,,,,',  // Empty row -- should fail validation
          'ORD-003,Bad Customer,,9876543210,Allocated-Printed,1',  // Missing address
        ].join('\n');

        await page.evaluate((csv) => {
          const input = document.querySelector('input[type="file"]') as HTMLInputElement;
          if (input) {
            const file = new File([csv], 'test-bad.csv', { type: 'text/csv' });
            const dt = new DataTransfer();
            dt.items.add(file);
            input.files = dt.files;
            input.dispatchEvent(new Event('change', { bubbles: true }));
          }
        }, badCSV);

        // Wait for results to appear (either error banner or import summary with failures)
        // The error table should appear if there are partial failures
        // Give extra time for geocoding etc.
        await page.waitForTimeout(5000);

        // Check if ErrorTable appeared (it shows when there are row-level failures)
        const errorTable = page.locator('[data-testid="error-table"]');
        const hasErrorTable = await errorTable.isVisible().catch(() => false);

        if (hasErrorTable) {
          // Verify table has the expected column headers
          await expect(errorTable.locator('th').first()).toBeVisible();

          // Verify at least one data row exists
          const rows = errorTable.locator('tbody tr');
          await expect(rows.first()).toBeVisible();
        } else {
          // If no error table, check that an ErrorBanner appeared instead
          // (the entire file may have been rejected)
          const banner = page.locator('[data-testid="error-banner"]');
          await expect(banner).toBeVisible({ timeout: 5_000 });
        }
      });

      // ─── Health Status Bar Tests ────────────────────────────

      test('Health status bar shows per-service status', async ({ page }) => {
        await page.goto('/dashboard/');
        await page.waitForLoadState('networkidle');

        // Wait for health polling to complete (polls every 30s, first poll on load)
        const healthBar = page.locator('[data-testid="health-status-bar"]');
        await expect(healthBar).toBeVisible({ timeout: 15_000 });

        // Verify it shows service names or status indicators
        // Should contain text like "operational", "healthy", or service names
        const healthText = await healthBar.textContent();
        expect(healthText).toBeTruthy();
        // Should mention at least one service or overall status
        expect(healthText).toMatch(/operational|healthy|postgresql|osrm|vroom|degraded|unhealthy/i);
      });

      // ─── No Console Errors ──────────────────────────────────

      test('No console errors during error UI interactions', async ({ page }) => {
        const consoleErrors: string[] = [];
        page.on('console', (msg) => {
          if (msg.type() === 'error') {
            consoleErrors.push(msg.text());
          }
        });

        await page.goto('/dashboard/');
        await page.waitForLoadState('networkidle');

        // Trigger an error
        await page.evaluate(() => {
          const input = document.querySelector('input[type="file"]') as HTMLInputElement;
          if (input) {
            const file = new File(['not a csv'], 'test.txt', { type: 'text/plain' });
            const dt = new DataTransfer();
            dt.items.add(file);
            input.files = dt.files;
            input.dispatchEvent(new Event('change', { bubbles: true }));
          }
        });

        // Wait for error banner
        await page.waitForTimeout(3000);

        // Filter out known benign errors (e.g., favicon 404)
        const realErrors = consoleErrors.filter(
          (e) => !e.includes('favicon') && !e.includes('sw.js')
        );

        expect(realErrors).toHaveLength(0);
      });
    });
    ```
