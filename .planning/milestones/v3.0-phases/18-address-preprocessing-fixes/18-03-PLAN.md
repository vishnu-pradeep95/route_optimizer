       ```typescript
       const zoneCircle = useMemo(() => {
         if (!zoneRadiusKm) return null;
         return circle(
           [VATAKARA_CENTER.longitude, VATAKARA_CENTER.latitude],
           zoneRadiusKm,
           { steps: 64, units: 'kilometers' }
         );
       }, [zoneRadiusKm]);
       ```

    4. Inside the `<Map>` component, BEFORE the route polylines (so the circle is behind routes), add:
       ```tsx
       {zoneCircle && (

         </Source>
       ```
